import React from 'react';

function ChatWidget({ agentUrl, width = '100%', height = '600px' }) {
  const iframeSrc = `https://your-app.azurewebsites.net/chat?agent=${encodeURIComponent(agentUrl)}`;
  
  return (
    <iframe
      src={iframeSrc}
      width={width}
      height={height}
      frameBorder="0"
      title="A2A Chat Widget"
    />
  );
}
<template>
  <iframe
    :src="iframeSrc"
    :width="width"
    :height="height"
    frameborder="0"
    title="A2A Chat Widget"
  ></iframe>
</template>

<script>
export default {
  props: {
    agentUrl: String,
    width: { type: String, default: '100%' },
    height: { type: String, default: '600px' }
  },
  computed: {
    iframeSrc() {
      return `https://your-app.azurewebsites.net/chat?agent=${encodeURIComponent(this.agentUrl)}`;
    }
  }
}
</script>
